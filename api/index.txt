<?php
header("Content-Type: application/json");
header("Access-Control-Allow-Origin: *");
header("Access-Control-Allow-Methods: GET,POST,PUT,PATCH,DELETE,OPTIONS");
header("Access-Control-Allow-Headers: Content-Type,Authorization");
if($_SERVER['REQUEST_METHOD']==='OPTIONS'){http_response_code(204);exit;}
$uri = explode("/",trim($_SERVER['REQUEST_URI'],"/"));
$endpoint = end($uri);

switch($endpoint){
    case "login": require "auth.php"; break;
    case "messages": require "endpoints/messages.php"; break;
    case "notes": require "endpoints/notes.php"; break;
    case "board": require "endpoints/board.php"; break;
    default: http_response_code(404); echo json_encode(["error"=>"Unknown endpoint"]);
}
